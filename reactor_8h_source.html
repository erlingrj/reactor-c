572struct trigger_t {
573 reaction_t** reactions; // Array of pointers to reactions sensitive to this trigger.
574 int number_of_reactions; // Number of reactions sensitive to this trigger.
575 bool is_timer; // True if this is a timer (a special kind of action), false otherwise.
576 interval_t offset; // Minimum delay of an action. For a timer, this is also the maximum delay.
577 interval_t period; // Minimum interarrival time of an action. For a timer, this is also the maximal interarrival time.
578 lf_token_t* token; // Pointer to a token wrapping the payload (or NULL if there is none).
579 bool is_physical; // Indicator that this denotes a physical action.
580 event_t* last; // Pointer to the last event that was scheduled for this action.
581 lf_spacing_policy_t policy; // Indicates which policy to use when an event is scheduled too early.
582 size_t element_size; // The size of the payload, if there is one, zero otherwise.
583 // If the payload is an array, then this is the size of an element of the array.
584 port_status_t status; // Determines the status of the port at the current logical time. Therefore, this
585 // value needs to be reset at the beginning of each logical time.
586 //
587 // This status is especially needed for the distributed execution because the receiver logic will need
588 // to know what it should do if it receives a message with 'intended tag = current tag' from another
589 // federate.
590 // - If status is 'unknown', it means that the federate has still no idea what the status of
591 // this port is and thus has refrained from executing any reaction that has that port as its input.
592 // This means that the receiver logic can directly inject the triggered reactions into the reaction
593 // queue at the current logical time.
594 // - If the status is absent, it means that the federate has assumed that the port is 'absent'
595 // for the current logical time. Therefore, receiving a message with 'intended tag = current tag'
596 // is an error that should be handled, for example, as a violation of the STP offset in the decentralized
597 // coordination.
598 // - Finally, if status is 'present', then this is an error since multiple
599 // downstream messages have been produced for the same port for the same logical time.
600 reactor_mode_t* mode; // The enclosing mode of this reaction (if exists).
601 // If enclosed in multiple, this will point to the innermost mode.
602#ifdef FEDERATED
603 tag_t last_known_status_tag; // Last known status of the port, either via a timed message, a port absent, or a
604 // TAG from the RTI.
605 bool is_a_control_reaction_waiting; // Indicates whether at least one control reaction is waiting for this trigger
606 // if it belongs to a network input port. Must be false by default.
607 tag_t intended_tag; // The amount of discrepency in logical time between the original intended
608 // trigger time of this trigger and the actual trigger time. This currently
609 // can only happen when logical connections are used using a decentralized coordination
610 // mechanism (@see https://github.com/icyphy/lingua-franca/wiki/Logical-Connections).
611 instant_t physical_time_of_arrival; // The physical time at which the message has been received on the network according to the local clock.
612 // Note: The physical_time_of_arrival is only passed down one level of the hierarchy. Default: NEVER.
613#endif
614};
