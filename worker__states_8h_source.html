82static size_t cond_of(size_t worker) {
83 // Note: __builtin_clz with GCC might be preferred, or fls (?).
84 int ret = 0;
85 while (worker) {
86 ret++;
87 worker >>= 1;
88 }
89 return ret;
90}

92static void worker_states_init(size_t number_of_workers) {
93 size_t greatest_worker_number = number_of_workers - 1;
94 size_t num_conds = cond_of(greatest_worker_number) + 1;
95 worker_conds = (lf_cond_t*) malloc(sizeof(lf_cond_t) * num_conds);
96 cumsum_of_worker_group_sizes = (size_t*) calloc(num_conds, sizeof(size_t));
97 mutex_held = (bool*) calloc(number_of_workers, sizeof(bool));
98 for (int i = 0; i < number_of_workers; i++) {
99 cumsum_of_worker_group_sizes[cond_of(i)]++;
100 }
101 for (int i = 1; i < num_conds; i++) {
102 cumsum_of_worker_group_sizes[i] += cumsum_of_worker_group_sizes[i - 1];
103 }
104 for (int i = 0; i < num_conds; i++) {
105 lf_cond_init(worker_conds + i, &mutex);
106 }
107 num_loose_threads = number_of_workers;
108}

123static void worker_states_awaken_locked(size_t worker, size_t num_to_awaken) {
124 assert(num_to_awaken <= max_num_workers);
125 if ((worker == 0) && (num_to_awaken <= 1)) {
126 num_loose_threads = 1;
127 return;
128 }
129 size_t greatest_worker_number_to_awaken = num_to_awaken - 1;
130 size_t max_cond = cond_of(greatest_worker_number_to_awaken);
131 if (!mutex_held[worker]) {
132 mutex_held[worker] = true;
133 lf_mutex_lock(&mutex);
134 }
135 // The predicate of the condition variable depends on num_awakened and level_counter, so
136 // this is a critical section.
137 num_loose_threads = cumsum_of_worker_group_sizes[max_cond];
138 num_loose_threads += worker >= num_loose_threads;
139 num_awakened = num_loose_threads;
140 level_counter++;
141 for (int cond = 0; cond <= max_cond; cond++) {
142 lf_cond_broadcast(worker_conds + cond);
143 }
144}

147static void worker_states_lock(size_t worker) {
148 assert(num_loose_threads > 0);
149 assert(num_loose_threads <= max_num_workers);
150 size_t lt = num_loose_threads;
151 if (lt > 1 || !fast) { // FIXME: Lock should be partially optimized out even when !fast
152 lf_mutex_lock(&mutex);
153 assert(!mutex_held[worker]);
154 mutex_held[worker] = true;
155 }
156}

159static void worker_states_unlock(size_t worker) {
160 if (!mutex_held[worker]) return;
161 mutex_held[worker] = false;
162 lf_mutex_unlock(&mutex);
163}

172static bool worker_states_finished_with_level_locked(size_t worker) {
173 assert(worker >= 0);
174 assert(num_loose_threads > 0);
175 assert(num_reactions_by_worker[worker] != 1);
176 assert(((int64_t) num_reactions_by_worker[worker]) <= 0);
177 // Why use an atomic operation when we are supposed to be "as good as locked"? Because I took a
178 // shortcut, and the shortcut was imperfect.
179 size_t ret = lf_atomic_add_fetch(&num_loose_threads, -1);
180 assert(ret <= max_num_workers); // Check for underflow
181 return !ret;
182}

194static void worker_states_sleep_and_unlock(size_t worker, size_t level_counter_snapshot) {
195 assert(worker < max_num_workers);
196 assert(num_loose_threads <= max_num_workers);
197 if (!mutex_held[worker]) {
198 lf_mutex_lock(&mutex);
199 }
200 mutex_held[worker] = false; // This will be true soon, upon call to lf_cond_wait.
201 size_t cond = cond_of(worker);
202 if (
203 ((level_counter_snapshot == level_counter) || worker >= num_awakened)
204 ) {
205 do {
206 lf_cond_wait(worker_conds + cond);
207 } while (level_counter_snapshot == level_counter || worker >= num_awakened);
208 }
209 assert(!mutex_held[worker]); // This thread holds the mutex, but it did not report that.
210 lf_mutex_unlock(&mutex);
211}
