144static int lf_cond_timedwait(lf_cond_t* cond, int64_t absolute_time_ns) {
145 // Convert the absolute time to a timespec.
146 // timespec is seconds and nanoseconds.
147 struct timespec timespec_absolute_time
148 = {(time_t)absolute_time_ns / 1000000000LL, (long)absolute_time_ns % 1000000000LL};
149 int return_value = 0;
150 return_value = cnd_timedwait(
151 (cnd_t*)&cond->condition,
152 (mtx_t*)cond->mutex,
153 &timespec_absolute_time
154 );
155 switch (return_value) {
156 case thrd_timedout:
157 return_value = _LF_TIMEOUT;
158 break;
159
160 default:
161 break;
162 }
163 return return_value;
164}
