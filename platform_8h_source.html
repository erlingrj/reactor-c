179/*
180 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
181 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
182 * @param value The value to be added to the variable pointed to by the ptr parameter.
183 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
184 */
185#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
186// Assume that an integer is 32 bits.
187#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
188#elif defined(__GNUC__) || defined(__clang__)
189#define lf_atomic_fetch_add(ptr, value) __sync_fetch_and_add(ptr, value)
190#else
191#error "Compiler not supported"
192#endif

196 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.

198 * @return The new value of the variable that ptr points to (i.e., from before the application of this operation).

209/*
210 * Atomically compare the variable that ptr points to against oldval. If the
211 * current value is oldval, then write newval into *ptr.
212 * @param ptr A pointer to a variable.
213 * @param oldval The value to compare against.
214 * @param newval The value to assign to *ptr if comparison is successful.
215 * @return True if comparison was successful. False otherwise.
216 */
217#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
218// Assume that a boolean is represented with a 32-bit integer.
219#define lf_bool_compare_and_swap(ptr, oldval, newval) (InterlockedCompareExchange(ptr, newval, oldval) == oldval)
220#elif defined(__GNUC__) || defined(__clang__)
221#define lf_bool_compare_and_swap(ptr, oldval, newval) __sync_bool_compare_and_swap(ptr, oldval, newval)
222#else
223#error "Compiler not supported"
224#endif
