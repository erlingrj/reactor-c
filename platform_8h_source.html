217/*
218 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
219 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
220 * @param value The value to be added to the variable pointed to by the ptr parameter.
221 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
222 */
223#if defined(PLATFORM_ZEPHYR)
224#define lf_atomic_fetch_add(ptr, value) _zephyr_atomic_fetch_add((int*) ptr, value)
225#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
226// Assume that an integer is 32 bits.
227#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
228#elif defined(__GNUC__) || defined(__clang__)
229#define lf_atomic_fetch_add(ptr, value) __sync_fetch_and_add(ptr, value)
230#else
231#error "Compiler not supported"
232#endif

251/*
252 * Atomically compare the variable that ptr points to against oldval. If the
253 * current value is oldval, then write newval into *ptr.
254 * @param ptr A pointer to a variable.
255 * @param oldval The value to compare against.
256 * @param newval The value to assign to *ptr if comparison is successful.
257 * @return True if comparison was successful. False otherwise.
258 */
259#if defined(PLATFORM_ZEPHYR)
260#define lf_bool_compare_and_swap(ptr, value, newval) _zephyr_bool_compare_and_swap((bool*) ptr, value, newval)
261#elif defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
262// Assume that a boolean is represented with a 32-bit integer.
263#define lf_bool_compare_and_swap(ptr, oldval, newval) (InterlockedCompareExchange(ptr, newval, oldval) == oldval)
264#elif defined(__GNUC__) || defined(__clang__)
265#define lf_bool_compare_and_swap(ptr, oldval, newval) __sync_bool_compare_and_swap(ptr, oldval, newval)
266#else
267#error "Compiler not supported"
268#endif
