177/*
178 * Atomically increment the variable that ptr points to by the given value, and return the original value of the variable.
179 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
180 * @param value The value to be added to the variable pointed to by the ptr parameter.
181 * @return The original value of the variable that ptr points to (i.e., from before the application of this operation).
182 */
183#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
184// Assume that an integer is 32 bits.
185#define lf_atomic_fetch_add(ptr, value) InterlockedExchangeAdd(ptr, value)
186#elif defined(__GNUC__) || defined(__clang__)
187#define lf_atomic_fetch_add(ptr, value) __sync_fetch_and_add(ptr, value)
188#else
189#error "Compiler not supported"
190#endif
191
192/*
193 * Atomically increment the variable that ptr points to by the given value, and return the new value of the variable.
194 * @param ptr A pointer to a variable. The value of this variable will be replaced with the result of the operation.
195 * @param value The value to be added to the variable pointed to by the ptr parameter.
196 * @return The new value of the variable that ptr points to (i.e., from before the application of this operation).
197 */
198#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
199// Assume that an integer is 32 bits.
200#define lf_atomic_add_fetch(ptr, value) InterlockedAdd(ptr, value)
201#elif defined(__GNUC__) || defined(__clang__)
202#define lf_atomic_add_fetch(ptr, value) __sync_add_and_fetch(ptr, value)
203#else
204#error "Compiler not supported"

207/*
208 * Atomically compare the variable that ptr points to against oldval. If the
209 * current value is oldval, then write newval into *ptr.
210 * @param ptr A pointer to a variable.
211 * @param oldval The value to compare against.
212 * @param newval The value to assign to *ptr if comparison is successful.
213 * @return The true if comparison was successful. False otherwise.
214 */
215#if defined(WIN32) || defined(_WIN32) || defined(__WIN32__) || defined(__NT__)
216// Assume that an integer is 32 bits.
217#define lf_bool_compare_and_swap(ptr, oldval, newval) (InterlockedCompareExchange(ptr, newval, oldval) == oldval)
218#elif defined(__GNUC__) || defined(__clang__)
219#define lf_bool_compare_and_swap(ptr, oldval, newval) __sync_bool_compare_and_swap(ptr, oldval, newval)
220#else
221#error "Compiler not supported"
