93typedef struct federate_t {
94 uint16_t id; // ID of this federate.
95 pthread_t thread_id; // The ID of the thread handling communication with this federate.
96 int socket; // The TCP socket descriptor for communicating with this federate.
97 struct sockaddr_in UDP_addr; // The UDP address for the federate.
98 bool clock_synchronization_enabled; // Indicates the status of clock synchronization
99 // for this federate. Enabled by default.
100 tag_t completed; // The largest logical tag completed by the federate (or NEVER if no LTC has been received).
101 tag_t last_granted; // The maximum TAG that has been granted so far (or NEVER if none granted)
102 tag_t last_provisionally_granted; // The maximum PTAG that has been provisionally granted (or NEVER if none granted)
103 tag_t next_event; // Most recent NET received from the federate (or NEVER if none received).
104 in_transit_message_record_q_t* in_transit_message_tags; // Record of in-transit messages to this federate that are not
105 // yet processed. This record is ordered based on the time
106 // value of each message for a more efficient access.
107 fed_state_t state; // State of the federate.
108 int* upstream; // Array of upstream federate ids.
109 interval_t* upstream_delay; // Minimum delay on connections from upstream federates.
110 // Here, NEVER encodes no delay. 0LL is a microstep delay.
111 int num_upstream; // Size of the array of upstream federates and delays.
112 int* downstream; // Array of downstream federate ids.
113 int num_downstream; // Size of the array of downstream federates.
114 execution_mode_t mode; // FAST or REALTIME.
115 char server_hostname[INET_ADDRSTRLEN]; // Human-readable IP address and
116 int32_t server_port; // port number of the socket server of the federate
117 // if it has any incoming direct connections from other federates.
118 // The port number will be -1 if there is no server or if the
119 // RTI has not been informed of the port number.
120 struct in_addr server_ip_addr; // Information about the IP address of the socket
121 // server of the federate.
122 bool requested_stop; // Indicates that the federate has requested stop or has replied
123 // to a request for stop from the RTI. Used to prevent double-counting
124 // a federate when handling lf_request_stop().
125} federate_t;
