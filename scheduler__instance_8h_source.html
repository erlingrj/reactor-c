150bool init_sched_instance(
151 _lf_sched_instance_t** instance,
152 size_t number_of_workers,
153 sched_params_t* params) {
154
155 // Check if the instance is already initialized
156 lf_mutex_lock(&mutex); // Safeguard against multiple threads calling this
157 // function.
158 if (*instance != NULL) {
159 // Already initialized
160 lf_mutex_unlock(&mutex);
161 return false;
162 } else {
163 *instance =
164 (_lf_sched_instance_t*)calloc(1, sizeof(_lf_sched_instance_t));
165 }
166 lf_mutex_unlock(&mutex);
167
168 if (params == NULL || params->num_reactions_per_level_size == 0) {
169 (*instance)->max_reaction_level = DEFAULT_MAX_REACTION_LEVEL;
170 }
171
172 if (params != NULL) {
173 if (params->num_reactions_per_level != NULL) {
174 (*instance)->max_reaction_level =
175 params->num_reactions_per_level_size - 1;
176 }
177 }
178
179 (*instance)->_lf_sched_semaphore = lf_semaphore_new(0);
180 (*instance)->_lf_sched_number_of_workers = number_of_workers;
181 (*instance)->_lf_sched_next_reaction_level = 1;
182
183 (*instance)->_lf_sched_should_stop = false;
184
185 return true;
186}
