216struct trigger_t {
217 token_template_t tmplt; // Type and token information (template is a C++ keyword).
218 reaction_t** reactions; // Array of pointers to reactions sensitive to this trigger.
219 int number_of_reactions; // Number of reactions sensitive to this trigger.
220 bool is_timer; // True if this is a timer (a special kind of action), false otherwise.
221 interval_t offset; // Minimum delay of an action. For a timer, this is also the maximum delay.
222 interval_t period; // Minimum interarrival time of an action. For a timer, this is also the maximal interarrival time.
223 bool is_physical; // Indicator that this denotes a physical action.
224 event_t* last; // Pointer to the last event that was scheduled for this action.
225 lf_spacing_policy_t policy; // Indicates which policy to use when an event is scheduled too early.
226 port_status_t status; // Determines the status of the port at the current logical time. Therefore, this
227 // value needs to be reset at the beginning of each logical time.
228 //
229 // This status is especially needed for the distributed execution because the receiver logic will need
230 // to know what it should do if it receives a message with 'intended tag = current tag' from another
231 // federate.
232 // - If status is 'unknown', it means that the federate has still no idea what the status of
233 // this port is and thus has refrained from executing any reaction that has that port as its input.
234 // This means that the receiver logic can directly inject the triggered reactions into the reaction
235 // queue at the current logical time.
236 // - If the status is absent, it means that the federate has assumed that the port is 'absent'
237 // for the current logical time. Therefore, receiving a message with 'intended tag = current tag'
238 // is an error that should be handled, for example, as a violation of the STP offset in the decentralized
239 // coordination.
240 // - Finally, if status is 'present', then this is an error since multiple
241 // downstream messages have been produced for the same port for the same logical time.
242 reactor_mode_t* mode; // The enclosing mode of this reaction (if exists).
243 // If enclosed in multiple, this will point to the innermost mode.
244#ifdef FEDERATED
245 tag_t last_known_status_tag; // Last known status of the port, either via a timed message, a port absent, or a
246 // TAG from the RTI.
247 bool is_a_control_reaction_waiting; // Indicates whether at least one control reaction is waiting for this trigger
248 // if it belongs to a network input port. Must be false by default.
249 tag_t intended_tag; // The amount of discrepency in logical time between the original intended
250 // trigger time of this trigger and the actual trigger time. This currently
251 // can only happen when logical connections are used using a decentralized coordination
252 // mechanism (@see https://github.com/icyphy/lingua-franca/wiki/Logical-Connections).
253 instant_t physical_time_of_arrival; // The physical time at which the message has been received on the network according to the local clock.
254 // Note: The physical_time_of_arrival is only passed down one level of the hierarchy. Default: NEVER.
255#endif
256};
