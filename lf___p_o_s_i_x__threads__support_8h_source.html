77static int lf_mutex_init(lf_mutex_t* mutex) {
78 // Set up a recursive mutex
79 pthread_mutexattr_t attr;
80 pthread_mutexattr_init(&attr);
81 // Initialize the mutex to be recursive, meaning that it is OK
82 // for the same thread to lock and unlock the mutex even if it already holds
83 // the lock.
84 // FIXME: This is dangerous. The docs say this: "It is advised that an
85 // application should not use a PTHREAD_MUTEX_RECURSIVE mutex with
86 // condition variables because the implicit unlock performed for a
87 // pthread_cond_wait() or pthread_cond_timedwait() may not actually
88 // release the mutex (if it had been locked multiple times).
89 // If this happens, no other thread can satisfy the condition
90 // of the predicate.” This seems like a bug in the implementation of
91 // pthreads. Maybe it has been fixed?
92 pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
93 return pthread_mutex_init((pthread_mutex_t*)mutex, &attr);
94}

119static int lf_cond_init(lf_cond_t* cond, lf_mutex_t* mutex) {
120 cond->mutex = mutex;
121 pthread_condattr_t cond_attr;
122 pthread_condattr_init(&cond_attr);
123 // Limit the scope of the condition variable to this process (default)
124 pthread_condattr_setpshared(&cond_attr, PTHREAD_PROCESS_PRIVATE);
125 return pthread_cond_init(&cond->condition, &cond_attr);
126}

164static int lf_cond_timedwait(lf_cond_t* cond, int64_t absolute_time_ns) {
165 // Convert the absolute time to a timespec.
166 // timespec is seconds and nanoseconds.
167 struct timespec timespec_absolute_time
168 = {(time_t)absolute_time_ns / 1000000000LL, (long)absolute_time_ns % 1000000000LL};
169 int return_value = 0;
170 return_value = pthread_cond_timedwait(
171 (pthread_cond_t*)&cond->condition,
172 (pthread_mutex_t*)cond->mutex,
173 &timespec_absolute_time
174 );
175 switch (return_value) {
176 case ETIMEDOUT:
177 return_value = _LF_TIMEOUT;
178 break;
179
180 default:
181 break;
182 }
183 return return_value;
184}
