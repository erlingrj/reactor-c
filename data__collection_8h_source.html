66static void possible_nums_workers_init() {
67 // Start with 0 and end with two numbers strictly greater than max_num_workers. This must start
68 // at 4 because the first two and last two entries are not counted.
69 size_t pnw_length = 4;
70 size_t temp = max_num_workers;
71 while ((temp >>= 1)) pnw_length++;
72 possible_nums_workers = (size_t*) malloc(pnw_length * sizeof(size_t));
73 temp = 1;
74 possible_nums_workers[0] = 0;
75 for (int i = 1; i < pnw_length; i++) {
76 possible_nums_workers[i] = temp;
77 temp *= 2;
78 }
79 assert(temp > max_num_workers);
80}

90static int get_jitter(size_t current_state, interval_t execution_time) {
91 static const size_t parallelism_cost_max = 114688;
92 // The following handles the case where the current level really is just fluff:
93 // No parallelism needed, no work to be done.
94 if (execution_time < 16384 && current_state == 1) return 0;
95 int left_score = 16384; // Want: For execution time = 65536, p(try left) = p(try right)
96 int middle_score = 65536;
97 int right_score = 65536;
98 if (execution_time < parallelism_cost_max) left_score += parallelism_cost_max - execution_time;
99 int result = rand() % (left_score + middle_score + right_score);
100 if (result < left_score) return -1;
101 if (result < left_score + middle_score) return 0;
102 return 1;
103}

106static size_t get_nums_workers_neighboring_state(size_t current_state, interval_t execution_time) {
107 size_t jitter = get_jitter(current_state, execution_time);
108 if (!jitter) return current_state;
109 size_t i = 1;
110 // TODO: There are more efficient ways to do this.
111 while (possible_nums_workers[i] < current_state) i++;
112 return possible_nums_workers[i + jitter];
113}

115static void data_collection_init(sched_params_t* params) {
116 size_t num_levels = params->num_reactions_per_level_size;
117 start_times_by_level = (interval_t*) calloc(num_levels, sizeof(interval_t));
118 execution_times_by_num_workers_by_level = (interval_t**) calloc(
119 num_levels, sizeof(interval_t*)
120 );
121 execution_times_mins = (interval_t*) calloc(num_levels, sizeof(interval_t));
122 execution_times_argmins = (size_t*) calloc(num_levels, sizeof(size_t));
123 for (size_t i = 0; i < num_levels; i++) {
124 execution_times_argmins[i] = max_num_workers;

126 max_num_workers + 1, // Add 1 for 1-based indexing

128 );
129 }
130 possible_nums_workers_init();

148static void data_collection_end_level(size_t level, size_t num_workers) {
149 if (collecting_data && start_times_by_level[level]) {
150 interval_t dt = lf_time_physical() - start_times_by_level[level];
151 if (!execution_times_by_num_workers_by_level[level][num_workers]) {
152 execution_times_by_num_workers_by_level[level][num_workers] = LF_MAX(
153 dt,
154 2 * execution_times_by_num_workers_by_level[level][execution_times_argmins[level]]
155 );
156 }
157 interval_t* prior_et = &execution_times_by_num_workers_by_level[level][num_workers];
158 *prior_et = (*prior_et * EXECUTION_TIME_MEMORY + dt) / (EXECUTION_TIME_MEMORY + 1);
159 }
160}

163 assert(start_inclusive <= end_inclusive);

177static void compute_number_of_workers(
178 size_t* num_workers_by_level,
179 size_t* max_num_workers_by_level,
180 bool jitter
181) {
182 for (size_t level = 0; level < num_levels; level++) {
183 interval_t this_execution_time = execution_times_by_num_workers_by_level[level][
184 num_workers_by_level[level]
185 ];
186 size_t ideal_number_of_workers;
187 size_t max_reasonable_num_workers = max_num_workers_by_level[level];
188 ideal_number_of_workers = execution_times_argmins[level];
189 int range = 1;
190 if (jitter) {
191 ideal_number_of_workers = get_nums_workers_neighboring_state(
192 ideal_number_of_workers, this_execution_time
193 );
194 }
195 int minimum_workers = 1;
196#ifdef WORKERS_NEEDED_FOR_FEDERATE
197 // TODO: only apply this constraint on levels containing control reactions
198 assert(WORKERS_NEEDED_FOR_FEDERATE >= 1);
199 minimum_workers = WORKERS_NEEDED_FOR_FEDERATE > max_reasonable_num_workers ?
200 max_reasonable_num_workers : WORKERS_NEEDED_FOR_FEDERATE;
201#endif
202 num_workers_by_level[level] = restrict_to_range(
203 minimum_workers, max_reasonable_num_workers, ideal_number_of_workers
204 );
205 }
206}

213static void compute_costs(size_t* num_workers_by_level) {
214 for (size_t level = 0; level < num_levels; level++) {
215 interval_t score = execution_times_by_num_workers_by_level[level][
216 num_workers_by_level[level]
217 ];
218 if (
219 !execution_times_mins[level]
220 | (score < execution_times_mins[level])
221 | (num_workers_by_level[level] == execution_times_argmins[level])
222 ) {
223 execution_times_mins[level] = score;
224 execution_times_argmins[level] = num_workers_by_level[level];
225 }
226 }
227}

235static void data_collection_end_tag(
236 size_t* num_workers_by_level,
237 size_t* max_num_workers_by_level
238) {
239 if (collecting_data && start_times_by_level[0]) {
240 compute_costs(num_workers_by_level);
241 }
242 data_collection_counter++;
243 size_t period = 2 + 128 * (data_collection_counter > SLOW_EXPERIMENTS);
244 size_t state = data_collection_counter % period;
245 if (state == 0) {
246 compute_number_of_workers(
247 num_workers_by_level,
248 max_num_workers_by_level,
249 data_collection_counter > START_EXPERIMENTS
250 );
251 collecting_data = true;
252 } else if (state == 1) {
253 compute_number_of_workers(num_workers_by_level, max_num_workers_by_level, false);
254 collecting_data = false;
255 }
256}
