63static void possible_nums_workers_init() {
64 // Start with 0 and end with two numbers strictly greater than max_num_workers. This must start
65 // at 4 because the first two and last two entries are not counted.
66 size_t pnw_length = 4;
67 size_t temp = max_num_workers;
68 while ((temp >>= 1)) pnw_length++;
69 possible_nums_workers = (size_t*) malloc(pnw_length * sizeof(size_t));
70 temp = 1;
71 possible_nums_workers[0] = 0;
72 for (int i = 1; i < pnw_length; i++) {
73 possible_nums_workers[i] = temp;
74 temp *= 2;
75 }
76 assert(temp > max_num_workers);
77}

87static int get_jitter(size_t current_state, interval_t execution_time) {
88 static const size_t parallelism_cost_max = 114688;
89 // The following handles the case where the current level really is just fluff:
90 // No parallelism needed, no work to be done.
91 if (execution_time < 16384 && current_state == 1) return 0;
92 int left_score = 16384; // Want: For execution time = 65536, p(try left) = p(try right)
93 int middle_score = 65536;
94 int right_score = 65536;
95 if (execution_time < parallelism_cost_max) left_score += parallelism_cost_max - execution_time;
96 int result = rand() % (left_score + middle_score + right_score);
97 if (result < left_score) return -1;
98 if (result < left_score + middle_score) return 0;
99 return 1;
100}

103static size_t get_nums_workers_neighboring_state(size_t current_state, interval_t execution_time) {
104 size_t jitter = get_jitter(current_state, execution_time);
105 if (!jitter) return current_state;
106 size_t i = 1;
107 // TODO: There are more efficient ways to do this.
108 while (possible_nums_workers[i] < current_state) i++;
109 return possible_nums_workers[i + jitter];
110}

114 start_times_by_level = (interval_t*) calloc(num_levels, sizeof(interval_t));
115 execution_times_by_num_workers_by_level = (interval_t**) calloc(

145static void data_collection_end_level(size_t level, size_t num_workers) {
146 if (collecting_data && start_times_by_level[level]) {
147 interval_t dt = lf_time_physical() - start_times_by_level[level];
148 if (!execution_times_by_num_workers_by_level[level][num_workers]) {
149 execution_times_by_num_workers_by_level[level][num_workers] = LF_MAX(
150 dt,
151 2 * execution_times_by_num_workers_by_level[level][execution_times_argmins[level]]
152 );
153 }
154 interval_t* prior_et = &execution_times_by_num_workers_by_level[level][num_workers];
155 *prior_et = (*prior_et * EXECUTION_TIME_MEMORY + dt) / (EXECUTION_TIME_MEMORY + 1);
156 }
157}

173static void compute_number_of_workers(
174 size_t* num_workers_by_level,
175 size_t* max_num_workers_by_level,
176 bool jitter
177) {
178 for (size_t level = 0; level < num_levels; level++) {
179 interval_t this_execution_time = execution_times_by_num_workers_by_level[level][
180 num_workers_by_level[level]
181 ];
182 size_t ideal_number_of_workers;
183 size_t max_reasonable_num_workers = max_num_workers_by_level[level];
184 ideal_number_of_workers = execution_times_argmins[level];
185 int range = 1;
186 if (jitter) {
187 ideal_number_of_workers = get_nums_workers_neighboring_state(
188 ideal_number_of_workers, this_execution_time
189 );
190 }
191 int minimum_workers = 1;
192#ifdef WORKERS_NEEDED_FOR_FEDERATE
193 // TODO: only apply this constraint on levels containing control reactions
194 minimum_workers = WORKERS_NEEDED_FOR_FEDERATE > max_reasonable_num_workers ?
195 max_reasonable_num_workers : WORKERS_NEEDED_FOR_FEDERATE;
196#endif
197 num_workers_by_level[level] = restrict_to_range(
198 minimum_workers, max_reasonable_num_workers, ideal_number_of_workers
199 );
200 }
201}

208static void compute_costs(size_t* num_workers_by_level) {
209 for (size_t level = 0; level < num_levels; level++) {
210 interval_t score = execution_times_by_num_workers_by_level[level][
211 num_workers_by_level[level]
212 ];
213 if (
214 !execution_times_mins[level]
215 | (score < execution_times_mins[level])
216 | (num_workers_by_level[level] == execution_times_argmins[level])
217 ) {
218 execution_times_mins[level] = score;
219 execution_times_argmins[level] = num_workers_by_level[level];
220 }
221 }
222}

230static void data_collection_end_tag(
231 size_t* num_workers_by_level,
232 size_t* max_num_workers_by_level
233) {
234 if (collecting_data && start_times_by_level[0]) {
235 compute_costs(num_workers_by_level);
236 }
237 data_collection_counter++;
238 size_t period = 2 + 128 * (data_collection_counter > SLOW_EXPERIMENTS);
239 size_t state = data_collection_counter % period;
240 if (state == 0) {
241 compute_number_of_workers(
242 num_workers_by_level,
243 max_num_workers_by_level,
244 data_collection_counter > START_EXPERIMENTS
245 );
246 collecting_data = true;
247 } else if (state == 1) {
248 compute_number_of_workers(num_workers_by_level, max_num_workers_by_level, false);
249 collecting_data = false;
250 }
251}
