72static void set_level(size_t level) {
73 assert(level < num_levels);
74 assert(0 <= level);
75 data_collection_end_level(current_level, num_workers);
76 current_level = level;
77 num_reactions_by_worker = num_reactions_by_worker_by_level[level];
78 reactions_by_worker = reactions_by_worker_by_level[level];
79 num_workers = num_workers_by_level[level];
80 // TODO: Experiment with not recording that the level is starting in the case that there is
81 // nothing to execute. We need not optimize for the case when there is nothing to execute
82 // because that case is not merely optimized, but is optimized out (we do not bother with
83 // executing nothing).
84 data_collection_start_level(current_level);
85}

97static void worker_assignments_init(size_t number_of_workers, sched_params_t* params) {
98 num_levels = params->num_reactions_per_level_size;
99 max_num_workers = number_of_workers;
100 reactions_by_worker_by_level = (reaction_t****) malloc(sizeof(reaction_t***) * num_levels);
101 num_reactions_by_worker_by_level = (size_t**) malloc(sizeof(size_t*) * num_levels);
102 num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
103 max_num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
104 for (size_t level = 0; level < num_levels; level++) {
105 size_t num_reactions = params->num_reactions_per_level[level];
106 size_t num_workers = num_reactions < max_num_workers ? num_reactions : max_num_workers;
107 max_num_workers_by_level[level] = num_workers;
108 num_workers_by_level[level] = max_num_workers_by_level[level];
109 reactions_by_worker_by_level[level] = (reaction_t***) malloc(
110 sizeof(reaction_t**) * max_num_workers
111 );
112 num_reactions_by_worker_by_level[level] = (size_t*) calloc(max_num_workers, sizeof(size_t));
113 for (size_t worker = 0; worker < max_num_workers_by_level[level]; worker++) {
114 reactions_by_worker_by_level[level][worker] = (reaction_t**) malloc(
115 sizeof(reaction_t*) * num_reactions
116 ); // Warning: This wastes space.
117 }
118 }
119 data_collection_init(params);
120 set_level(0);
121}

141static reaction_t* get_reaction(size_t worker) {
142#ifndef FEDERATED
143 int index = lf_atomic_add_fetch(num_reactions_by_worker + worker, -1);
144 if (index >= 0) {
145 return reactions_by_worker[worker][index];
146 }
147 num_reactions_by_worker[worker] = 0;
148 return NULL;
149#else
150 // This is necessary for federated programs because reactions may be inserted into the current
151 // level.
152 int old_num_reactions;
153 int current_num_reactions = num_reactions_by_worker[worker];
154 int index;
155 do {
156 old_num_reactions = current_num_reactions;
157 if (old_num_reactions <= 0) return NULL;
158 } while (
159 (current_num_reactions = lf_val_compare_and_swap(
160 num_reactions_by_worker + worker,
161 old_num_reactions,
162 (index = old_num_reactions - 1)
163 )) != old_num_reactions
164 );
165 return reactions_by_worker[worker][index];
166#endif
167}

175static reaction_t* worker_assignments_get_or_lock(size_t worker) {
176 assert(worker >= 0);
177 // assert(worker < num_workers); // There are edge cases where this doesn't hold.
178 assert(num_reactions_by_worker[worker] >= 0);
179 reaction_t* ret;
180 while (true) {
181 if ((ret = get_reaction(worker))) return ret;
182 if (worker < num_workers) {
183 for (
184 size_t victim = (worker + 1) % num_workers;
185 victim != worker;
186 victim = (victim + 1) % num_workers
187 ) {
188 if ((ret = get_reaction(victim))) return ret;
189 }
190 }
191 worker_states_lock(worker);
192 if (!num_reactions_by_worker[worker]) {
193 return NULL;
194 }
195 worker_states_unlock(worker);
196 }
197}

203static void worker_assignments_put(reaction_t* reaction) {
204 size_t level = LEVEL(reaction->index);
205 assert(reaction != NULL);
206#ifndef FEDERATED
207 assert(level > current_level || current_level == 0);
208#endif
209 assert(level < num_levels);
210 // Source: https://xorshift.di.unimi.it/splitmix64.c
211 // TODO: This is probably not the most efficient way to get the randomness that we need because
212 // it is designed to give an entire word of randomness, whereas we only need
213 // ~log2(num_workers_by_level[level]) bits of randomness.
214 uint64_t hash = (uint64_t) reaction;
215 hash = (hash ^ (hash >> 30)) * 0xbf58476d1ce4e5b9;
216 hash = (hash ^ (hash >> 27)) * 0x94d049bb133111eb;
217 hash = hash ^ (hash >> 31);
218 size_t worker = hash % num_workers_by_level[level];
219 size_t num_preceding_reactions = lf_atomic_fetch_add(
220 &num_reactions_by_worker_by_level[level][worker],
221 1
222 );
223 reactions_by_worker_by_level[level][worker][num_preceding_reactions] = reaction;
224}
