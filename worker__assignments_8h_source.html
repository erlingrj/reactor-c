75static void set_level(size_t level) {

77 assert(0 <= level);
78 data_collection_end_level(current_level, num_workers);

100static void worker_assignments_init(size_t number_of_workers, sched_params_t* params) {
101 num_levels = params->num_reactions_per_level_size;
102 max_num_workers = number_of_workers;
103 reactions_by_worker_by_level = (reaction_t****) malloc(sizeof(reaction_t***) * num_levels);
104 num_reactions_by_worker_by_level = (size_t**) malloc(sizeof(size_t*) * num_levels);
105 num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
106 max_num_workers_by_level = (size_t*) malloc(sizeof(size_t) * num_levels);
107 for (size_t level = 0; level < num_levels; level++) {
108 size_t num_reactions = params->num_reactions_per_level[level];
109 size_t num_workers = num_reactions < max_num_workers ? num_reactions : max_num_workers;
110 max_num_workers_by_level[level] = num_workers;
111 num_workers_by_level[level] = max_num_workers_by_level[level];
112 reactions_by_worker_by_level[level] = (reaction_t***) malloc(
113 sizeof(reaction_t**) * max_num_workers
114 );
115 num_reactions_by_worker_by_level[level] = (size_t*) calloc(max_num_workers, sizeof(size_t));
116 for (size_t worker = 0; worker < max_num_workers_by_level[level]; worker++) {
117 reactions_by_worker_by_level[level][worker] = (reaction_t**) malloc(
118 sizeof(reaction_t*) * num_reactions
119 ); // Warning: This wastes space.
120 }
121 }
122 data_collection_init(params);
123 set_level(0);
124}

144static reaction_t* get_reaction(size_t worker) {
145#ifndef FEDERATED
146 int index = lf_atomic_add_fetch(num_reactions_by_worker + worker, -1);
147 if (index >= 0) {
148 return reactions_by_worker[worker][index];
149 }
150 num_reactions_by_worker[worker] = 0;
151 return NULL;
152#else
153 // This is necessary for federated programs because reactions may be inserted into the current
154 // level.
155 int old_num_reactions;
156 int current_num_reactions = num_reactions_by_worker[worker];
157 int index;
158 do {
159 old_num_reactions = current_num_reactions;
160 if (old_num_reactions <= 0) return NULL;
161 } while (
162 (current_num_reactions = lf_val_compare_and_swap(
163 num_reactions_by_worker + worker,
164 old_num_reactions,
165 (index = old_num_reactions - 1)
166 )) != old_num_reactions
167 );
168 return reactions_by_worker[worker][index];
169#endif
170}

178static reaction_t* worker_assignments_get_or_lock(size_t worker) {
179 assert(worker >= 0);
180 // assert(worker < num_workers); // There are edge cases where this doesn't hold.
181 assert(num_reactions_by_worker[worker] >= 0);
182 reaction_t* ret;
183 while (true) {
184 if ((ret = get_reaction(worker))) return ret;
185 if (worker < num_workers) {
186 for (
187 size_t victim = (worker + 1) % num_workers;
188 victim != worker;
189 victim = (victim + 1) % num_workers
190 ) {
191 if ((ret = get_reaction(victim))) return ret;
192 }
193 }
194 worker_states_lock(worker);
195 if (!num_reactions_by_worker[worker]) {
196 return NULL;
197 }
198 worker_states_unlock(worker);
199 }
200}

206static void worker_assignments_put(reaction_t* reaction) {
207 size_t level = LF_LEVEL(reaction->index);
208 assert(reaction != NULL);
209#ifndef FEDERATED
210 assert(level > current_level || current_level == 0);
211#endif
212 assert(level < num_levels);
213 // Source: https://xorshift.di.unimi.it/splitmix64.c
214 // TODO: This is probably not the most efficient way to get the randomness that we need because
215 // it is designed to give an entire word of randomness, whereas we only need
216 // ~log2(num_workers_by_level[level]) bits of randomness.
217 uint64_t hash = (uint64_t) reaction;
218 hash = (hash ^ (hash >> 30)) * 0xbf58476d1ce4e5b9;
219 hash = (hash ^ (hash >> 27)) * 0x94d049bb133111eb;
220 hash = hash ^ (hash >> 31);
221 size_t worker = hash % num_workers_by_level[level];
222 size_t num_preceding_reactions = lf_atomic_fetch_add(
223 &num_reactions_by_worker_by_level[level][worker],
224 1
225 );
226 reactions_by_worker_by_level[level][worker][num_preceding_reactions] = reaction;
227}
